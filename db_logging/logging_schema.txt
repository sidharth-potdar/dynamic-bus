CREATE TABLE Requests (
	id INTEGER PRIMARY KEY,
	ride_id INTEGER,
	sim_id INTEGER,
	execute_ts NUMERIC,
	wallclock_ts TEXT,
	origin_node NUMERIC,
	dest_node NUMERIC,
	priority NUMERIC,
	FOREIGN KEY (sim_id) REFERENCES SimRuns(sim_id)
);

CREATE TABLE Schedules (
	id INTEGER PRIMARY KEY,
	ride_id INTEGER,
	sim_id INTEGER,
	execute_ts NUMERIC,
	wallclock_ts TEXT,
	origin_node NUMERIC,
	dest_node NUMERIC,
	priority NUMERIC,
	bus_id INTEGER,
	FOREIGN KEY (sim_id) REFERENCES SimRuns(sim_id)
);

CREATE TABLE Pickups (
	id INTEGER PRIMARY KEY,
	ride_id INTEGER,
	sim_id INTEGER,
	execute_ts NUMERIC,
	wallclock_ts TEXT,
	location NUMERIC,
	priority NUMERIC,
	bus_id INTEGER,
	FOREIGN KEY (sim_id) REFERENCES SimRuns(sim_id)
);

CREATE TABLE Dropoffs (
	id INTEGER PRIMARY KEY,
	ride_id INTEGER,
	sim_id INTEGER,
	execute_ts NUMERIC,
	wallclock_ts TEXT,
	location NUMERIC,
	priority NUMERIC,
	bus_id INTEGER
	FOREIGN KEY (sim_id) REFERENCES SimRuns(sim_id)
);

CREATE TABLE SimRuns (
	sim_id INTEGER PRIMARY KEY,
	num_rides INTEGER,
	num_buses INTEGER,
	bus_capacity INTEGER,
	execution_time NUMERIC
);
